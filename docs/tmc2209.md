/*
 * Pompa perystaltyczna - ESP32 XIAO C3 + TMC2209 (standalone, bez UART)
 * STEP generowany sprzętowo przez LEDC (PWM).
 *
 * POŁĄCZENIA:
 *   GPIO2  (D0) → EN    (LOW = aktywny)
 *   GPIO7  (D5) → STEP
 *   GPIO21 (D6) → DIR
 *   GPIO3  (D1) → MS1   (lub podłącz do GND/3V3 na stałe)
 *   GPIO4  (D2) → MS2   (lub podłącz do GND/3V3 na stałe)
 *   GPIO5  (D3) → DIAG  (HIGH = stall)
 *
 * MS1/MS2 tabela (TMC2209 standalone):
 *   MS2=0 MS1=0 → 1/8   StealthChop2
 *   MS2=0 MS1=1 → 1/2   StealthChop2
 *   MS2=1 MS1=0 → 1/4   StealthChop2
 *   MS2=1 MS1=1 → 1/16  SpreadCycle
 */

#include <Arduino.h>
#include <WiFi.h>

// ===== PINY =====
#define STEP_PIN    7
#define DIR_PIN     21
#define ENABLE_PIN  2
#define MS1_PIN     3
#define MS2_PIN     4
#define DIAG_PIN    5

// ===== KONFIGURACJA RUCHU =====
// Liczba impulsów STEP na sekundę → bezpośrednio częstotliwość LEDC
// Przelicznik: SPS = RPM * STEPS_PER_REV * MICROSTEP_DIV / 60
// Przykład: 100 RPM, 200 krok/obr, 1/8 → 100*200*8/60 ≈ 2667 SPS
#define STEPS_PER_SEC   4000U   // kroki/sekundę

#define DIR_CW          HIGH    // kierunek: CW = HIGH, CCW = LOW

// Mikrokrokowanie — ustaw MS1/MS2 zgodnie z tabelą powyżej
// 0 = LOW, 1 = HIGH
#define MS1_STATE       LOW     // 1/8: MS1=0, MS2=0
#define MS2_STATE       LOW

// ===== LEDC =====
#define LEDC_RESOLUTION 10      // 10-bit → zakres duty 0–1023
#define LEDC_DUTY_50    512     // 50% duty cycle

// ============================================================
void setup() {
  WiFi.mode(WIFI_OFF);
  btStop();

  Serial.begin(115200);

  // Konfiguracja pinów
  pinMode(DIR_PIN,    OUTPUT);
  pinMode(ENABLE_PIN, OUTPUT);
  pinMode(MS1_PIN,    OUTPUT);
  pinMode(MS2_PIN,    OUTPUT);
  pinMode(DIAG_PIN,   INPUT);

  digitalWrite(DIR_PIN,    DIR_CW);
  digitalWrite(ENABLE_PIN, HIGH);    // silnik wyłączony do startu
  digitalWrite(MS1_PIN,    MS1_STATE);
  digitalWrite(MS2_PIN,    MS2_STATE);

  // LEDC: podepnij pin STEP (API core 3.x)
  ledcAttach(STEP_PIN, STEPS_PER_SEC, LEDC_RESOLUTION);
  ledcWrite(STEP_PIN, 0);            // zatrzymany na starcie

  // Włącz silnik
  digitalWrite(ENABLE_PIN, LOW);
  ledcWrite(STEP_PIN, LEDC_DUTY_50);

  Serial.println("START: " + String(STEPS_PER_SEC) + " SPS");
}

// ============================================================
void loop() {
  // Stall detection: DIAG HIGH = zablokowanie / błąd
  static bool stallPrev = false;
  bool stall = digitalRead(DIAG_PIN);
  if (stall && !stallPrev) {
    Serial.println("[ALARM] STALL!");
  }
  stallPrev = stall;
}
